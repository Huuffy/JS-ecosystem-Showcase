  const subdomain = req.headers.host.split('.')[0];

function DocumentUpload() {
  const [uploadProgress, setUploadProgress] = useState(0);
  const [dragActive, setDragActive] = useState(false);
  
  const handleDrop = (e) => {
    e.preventDefault();
    setDragActive(false);
    const files = e.dataTransfer.files;
    uploadFiles(files);
  };
  
  const uploadFiles = async (files) => {
    const formData = new FormData();
    Array.from(files).forEach(file => formData.append('documents', file));
    
    const xhr = new XMLHttpRequest();
    xhr.upload.onprogress = (e) => {
      setUploadProgress((e.loaded / e.total) * 100);
    };
    
    xhr.onload = () => {
      if (xhr.status === 200) {
        const response = JSON.parse(xhr.responseText);
        onUploadComplete(response.documents);
      }
    };
    
    xhr.open('POST', '/api/documents/upload');
    xhr.send(formData);
  };
}

const sharp = require('sharp');
const pdf2pic = require('pdf2pic');

const processDocument = async (fileId) => {
  const file = await gfs.files.findOne({ _id: fileId });
  
  if (file.contentType.includes('image')) {
    // Generate thumbnails for images
    const readStream = gfs.createReadStream({ _id: fileId });
    const thumbnail = await sharp(readStream)
      .resize(200, 200)
      .jpeg({ quality: 80 })
      .toBuffer();
    
    // Save thumbnail to GridFS
    const writeStream = gfs.createWriteStream({
      filename: `thumb_${file.filename}`,
      metadata: { originalId: fileId, type: 'thumbnail' }
    });
    writeStream.write(thumbnail);
    writeStream.end();
  }
  
  if (file.contentType === 'application/pdf') {
    // Extract text for search indexing
    const pdfText = await extractPDFText(fileId);
    await updateDocumentMetadata(fileId, { searchText: pdfText });
  }
};

// MongoDB Aggregation for Cross-Department Analytics
const getDepartmentMetrics = async (department, dateRange) => {
  return await Employee.aggregate([
    { $match: { department, createdAt: { $gte: dateRange.start, $lte: dateRange.end }}},
    {
      $lookup: {
        from: 'projects',
        localField: '_id',
        foreignField: 'assignedTo',
        as: 'projects'
      }
    },
    {
      $lookup: {
        from: 'expenses',
        localField: '_id',
        foreignField: 'employeeId',
        as: 'expenses'
      }
    },
    {
      $group: {
        _id: '$department',
        totalEmployees: { $sum: 1 },
        activeProjects: { $sum: { $size: '$projects' }},
        totalExpenses: { $sum: { $sum: '$expenses.amount' }},
        avgSalary: { $avg: '$salary' }
      }
    }
  ]);
};

// Express.js Role-Based Access Control
const roleMiddleware = (requiredRoles) => {
  return (req, res, next) => {
    const userRoles = req.user.roles;
    const hasPermission = requiredRoles.some(role => userRoles.includes(role));
    
    if (!hasPermission) {
      return res.status(403).json({ error: 'Insufficient permissions' });
    }
    next();
  };
};

// Node.js Cryptocurrency Payment Processing
const bitcoin = require('bitcoinjs-lib');
const axios = require('axios');

const processPayment = async (orderId, amount, recipientAddress) => {
  try {
    // Create Bitcoin transaction
    const network = bitcoin.networks.testnet;
    const keyPair = bitcoin.ECPair.fromWIF(process.env.BITCOIN_PRIVATE_KEY, network);
    const { address } = bitcoin.payments.p2pkh({ pubkey: keyPair.publicKey, network });
    
    // Get UTXOs
    const utxos = await getUTXOs(address);
    const txb = new bitcoin.TransactionBuilder(network);
    
    let totalInput = 0;
    utxos.forEach(utxo => {
      txb.addInput(utxo.txid, utxo.vout);
      totalInput += utxo.value;
    });
    
    const fee = 10000; // 0.0001 BTC
    const change = totalInput - amount - fee;
    
    txb.addOutput(recipientAddress, amount);
    if (change > 0) {
      txb.addOutput(address, change);
    }
    
    // Sign transaction
    utxos.forEach((utxo, index) => {
      txb.sign(index, keyPair);
    });
    
    const tx = txb.build();
    const txHex = tx.toHex();
    
    // Broadcast transaction
    const response = await axios.post('https://api.blockcypher.com/v1/btc/test3/txs/push', {
      tx: txHex
    });
    
    // Update order with payment info
    await Order.findByIdAndUpdate(orderId, {
      paymentStatus: 'completed',
      paymentTxHash: response.data.tx.hash,
      paidAt: new Date()
    });
    
    return response.data.tx.hash;
  } catch (error) {
    console.error('Payment processing failed:', error);
    throw error;
  }
};
